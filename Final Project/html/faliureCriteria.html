            if (superimposedParam.locStress(1, 1, jj) >= strength.X) || ...

                % Does laminate fail?
                fail{jj} = 'Yes';

                % What is the mode of faliure?
                mode{jj} = 'Fiber Faliure';


                % Checking for matrix faliure (Method 1)

                fail{jj} = 'Yes';
